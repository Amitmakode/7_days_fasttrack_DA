Yeh file ek **Retail Store ka OLTP Database Schema** hai.

Isme customers, products, stores, suppliers, sales orders aur inventory ke tables 
banaye gaye hain — basically ek retail business ke din-badin transactions store karne ka structure.

**OLTP** (Online Transaction Processing) isliye use kiya kyunki yeh real-time data insert/update
 ke liye optimized hota hai — jaise koi order place kare, stock ghate, ya customer register ho.

Indexes aur foreign key relationships isliye diye hain taaki queries fast rahe aur data 
consistent rahe.


**Nahi**, OLTP raw database nahi hota — yeh **khud primary database hota hai** jisme 
live transactions directly store hote hain.

Isse **Database Architects ya Backend Developers** banate hain, business ke requirements
 ke hisaab se.

Iska purpose hota hai **daily operations** handle karna — jaise orders, payments, 
inventory updates — real-time mein fast insert/update ke liye.

Baad mein isi OLTP data se **OLAP / Data Warehouse** banaya jata hai jo reporting aur 
analytics ke liye hota hai.

-- 1. ROW_NUMBER, RANK, DENSE_RANK

-- Live Query 1 — Store Ranking by Total Sales

SELECT 
    ds.store_name,
    ds.city,
    ds.store_type,
    SUM(fs.sale_amount) AS total_sales,
    ROW_NUMBER() OVER (ORDER BY SUM(fs.sale_amount) DESC) AS row_num,
    RANK()       OVER (ORDER BY SUM(fs.sale_amount) DESC) AS rank_num,
    DENSE_RANK() OVER (ORDER BY SUM(fs.sale_amount) DESC) AS dense_rank_num
FROM retail_rolap.fact_sales fs
JOIN retail_rolap.dim_store ds ON fs.store_key = ds.store_key
GROUP BY ds.store_key, ds.store_name, ds.city, ds.store_type
ORDER BY total_sales DESC;


-- Live Query 2 — Top 3 Products EACH Category mein


SELECT * FROM (
    SELECT 
        dp.category,
        dp.product_name,
        SUM(fs.sale_amount) AS revenue,
        DENSE_RANK() OVER (
            PARTITION BY dp.category 
            ORDER BY SUM(fs.sale_amount) DESC
        ) AS rank_in_category
    FROM retail_rolap.fact_sales fs
    JOIN retail_rolap.dim_product dp ON fs.product_key = dp.product_key
    GROUP BY dp.category, dp.product_key, dp.product_name
) ranked
WHERE rank_in_category <= 3;


-- 2. LAG aur LEAD

-- LAG → pichli row ki value lao (previous month sales)
-- LEAD → agli row ki value lao (next month sales)

-- Live Query 3 — Month-over-Month Sales Comparison

SELECT 
    `year_month`,
    category,
    total_sales,
    LAG(total_sales)  OVER (PARTITION BY category ORDER BY `year_month`) AS prev_month_sales,
    LEAD(total_sales) OVER (PARTITION BY category ORDER BY `year_month`) AS next_month_sales,
    ROUND(
        (total_sales - LAG(total_sales) OVER (PARTITION BY category ORDER BY `year_month`))
        / NULLIF(LAG(total_sales) OVER (PARTITION BY category ORDER BY `year_month`), 0) * 100
    , 2) AS mom_growth_pct
FROM retail_rolap.agg_monthly_category_sales
ORDER BY category, `year_month`;

store_ranked AS (

        DENSE_RANK() OVER (PARTITION BY store_type ORDER BY total_sales DESC) AS rank_in_type
    FROM store_sales

SELECT * FROM store_ranked

ORDER BY store_type, rank_in_type;

-- 5. 10 DMart SQL Problems

-- Q1. Har state mein sabse zyada sales wala store kaun sa hai?

WITH state_store AS (
    SELECT ds.state, ds.store_name,
        SUM(fs.sale_amount) AS total_sales,
        RANK() OVER (PARTITION BY ds.state ORDER BY SUM(fs.sale_amount) DESC) AS rnk
    FROM fact_sales fs
    JOIN dim_store ds ON fs.store_key = ds.store_key
    GROUP BY ds.state, ds.store_key, ds.store_name
)
SELECT state, store_name, total_sales
FROM state_store WHERE rnk = 1;


-- Q2. Top 5 products jo last 3 months mein sabse zyada biki?

SELECT 
    dp.product_name, dp.category,
    SUM(fs.quantity) AS units_sold,
    SUM(fs.sale_amount) AS revenue
FROM fact_sales fs
JOIN dim_product dp ON fs.product_key = dp.product_key
WHERE fs.date_id >= DATE_FORMAT(DATE_SUB(CURDATE(), INTERVAL 3 MONTH), '%Y%m%d')
GROUP BY dp.product_key, dp.product_name, dp.category
ORDER BY units_sold DESC
LIMIT 5;


-- Q3. Har month ka best selling category kaun sa raha?

WITH monthly_cat AS (
    SELECT 
        d.year_val, d.month_name,
        dp.category,
        SUM(fs.sale_amount) AS total_sales,
        RANK() OVER (PARTITION BY d.year_val, d.month_val ORDER BY SUM(fs.sale_amount) DESC) AS rnk
    FROM fact_sales fs
    JOIN dim_date d ON fs.date_id = d.date_id
    JOIN dim_product dp ON fs.product_key = dp.product_key
    GROUP BY d.year_val, d.month_val, d.month_name, dp.category
)
SELECT year_val, month_name, category, total_sales
FROM monthly_cat WHERE rnk = 1
ORDER BY year_val, total_sales DESC;


-- Q4. VIP customers jinhe 60+ din se kuch nahi liya (churn risk)?

SELECT 
    dc.customer_name, dc.city,
    MAX(d.full_date) AS last_purchase,
    DATEDIFF(CURDATE(), MAX(d.full_date)) AS days_inactive,
    SUM(fs.sale_amount) AS lifetime_value
FROM dim_customer dc
JOIN fact_sales fs ON dc.customer_key = fs.customer_key
JOIN dim_date d ON fs.date_id = d.date_id
WHERE dc.customer_segment = 'VIP'
GROUP BY dc.customer_key, dc.customer_name, dc.city
HAVING days_inactive > 60
ORDER BY days_inactive DESC;


-- Q5. Har store ki MoM (Month over Month) growth?

WITH monthly AS (
    SELECT 
        fs.store_key,
        d.year_val * 100 + d.month_val AS `year_month`,
        SUM(fs.sale_amount) AS total_sales
    FROM fact_sales fs
    JOIN dim_date d ON fs.date_id = d.date_id
    GROUP BY fs.store_key, d.year_val, d.month_val
)
SELECT 
    ds.store_name,
    m.`year_month`,
    m.total_sales,
    LAG(m.total_sales) OVER (PARTITION BY m.store_key ORDER BY m.`year_month`) AS prev_month,
    ROUND(
        (m.total_sales - LAG(m.total_sales) OVER (PARTITION BY m.store_key ORDER BY m.`year_month`))
        / NULLIF(LAG(m.total_sales) OVER (PARTITION BY m.store_key ORDER BY m.`year_month`), 0) * 100
    , 2) AS mom_growth
FROM monthly m
JOIN dim_store ds ON m.store_key = ds.store_key
ORDER BY ds.store_name, m.`year_month`;

-- Q6. Sabse zyada profitable brand kaun sa hai?

SELECT 
    dp.brand,
    SUM(fs.sale_amount) AS revenue,
    SUM(fs.profit) AS total_profit,
    ROUND(SUM(fs.profit) / NULLIF(SUM(fs.sale_amount), 0) * 100, 2) AS profit_margin_pct
FROM fact_sales fs
JOIN dim_product dp ON fs.product_key = dp.product_key
GROUP BY dp.brand
ORDER BY total_profit DESC
LIMIT 10;



-- Q7. Weekend vs Weekday sales comparison?

SELECT 
    CASE WHEN d.is_weekend = 1 THEN 'Weekend' ELSE 'Weekday' END AS day_type,
    COUNT(DISTINCT fs.transaction_id) AS total_orders,
    SUM(fs.sale_amount) AS total_sales,
    ROUND(AVG(fs.sale_amount), 2) AS avg_order_value
FROM fact_sales fs
JOIN dim_date d ON fs.date_id = d.date_id
GROUP BY d.is_weekend;


-- Q8. City tier wise customer spending pattern?

SELECT 
    dc.city_tier,
    dc.customer_segment,
    COUNT(DISTINCT dc.customer_key) AS customers,
    ROUND(AVG(fs.sale_amount), 2) AS avg_order_value,
    SUM(fs.sale_amount) AS total_sales
FROM dim_customer dc
JOIN fact_sales fs ON dc.customer_key = fs.customer_key
GROUP BY dc.city_tier, dc.customer_segment
ORDER BY dc.city_tier, total_sales DESC;



-- Q9. Har product ki running total sales (cumulative)?

SELECT 
    d.full_date,
    dp.product_name,
    SUM(fs.sale_amount) AS daily_sales,
    SUM(SUM(fs.sale_amount)) OVER (
        PARTITION BY dp.product_key 
        ORDER BY d.full_date
        ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW
    ) AS cumulative_sales
FROM fact_sales fs
JOIN dim_date d ON fs.date_id = d.date_id
JOIN dim_product dp ON fs.product_key = dp.product_key
GROUP BY d.full_date, dp.product_key, dp.product_name
ORDER BY dp.product_name, d.full_date;


-- Q10. Loyalty vs Non-loyalty customers ka revenue difference?

SELECT 
    CASE WHEN dc.loyalty_member = 1 THEN 'Loyalty Member' ELSE 'Non-Loyalty' END AS member_type,
    COUNT(DISTINCT dc.customer_key) AS customer_count,
    SUM(fs.sale_amount) AS total_revenue,
    ROUND(AVG(fs.sale_amount), 2) AS avg_order_value,
    ROUND(SUM(fs.sale_amount) / COUNT(DISTINCT dc.customer_key), 2) AS revenue_per_customer
FROM dim_customer dc
JOIN fact_sales fs ON dc.customer_key = fs.customer_key
GROUP BY dc.loyalty_member;

-- Query — Category wise MoM Sales Comparison

WITH monthly_sales AS (
    SELECT 
        dp.category,
        d.year_val,
        d.month_val,
        d.year_val * 100 + d.month_val AS `year_month`,
        d.month_name,
        SUM(fs.sale_amount) AS total_sales,
        SUM(fs.profit) AS total_profit,
        COUNT(DISTINCT fs.transaction_id) AS total_orders
    FROM fact_sales fs
    JOIN dim_date d ON fs.date_id = d.date_id
    JOIN dim_product dp ON fs.product_key = dp.product_key
    GROUP BY dp.category, d.year_val, d.month_val, d.month_name
),
mom_comparison AS (
    SELECT 
        category,
        `year_month`,
        month_name,
        year_val,
        total_sales,
        total_profit,
        total_orders,

        -- Pichle mahine ki sales
        LAG(total_sales) OVER (
            PARTITION BY category ORDER BY `year_month`
        ) AS prev_month_sales,

        -- Sales difference
        total_sales - LAG(total_sales) OVER (
            PARTITION BY category ORDER BY `year_month`
        ) AS sales_difference,

        -- MoM Growth %
        ROUND(
            (total_sales - LAG(total_sales) OVER (PARTITION BY category ORDER BY `year_month`))
            / NULLIF(LAG(total_sales) OVER (PARTITION BY category ORDER BY `year_month`), 0) * 100
        , 2) AS mom_growth_pct,

        -- Trend indicator
        CASE 
            WHEN total_sales > LAG(total_sales) OVER (PARTITION BY category ORDER BY `year_month`)
                THEN 'UP'
            WHEN total_sales < LAG(total_sales) OVER (PARTITION BY category ORDER BY `year_month`)
                THEN 'DOWN'
            ELSE 'FLAT'
        END AS trend
    FROM monthly_sales
)
SELECT 
    category,
    `year_month`,
    month_name,
    year_val,
    ROUND(total_sales, 2)       AS current_sales,
    ROUND(prev_month_sales, 2)  AS prev_month_sales,
    ROUND(sales_difference, 2)  AS sales_difference,
    mom_growth_pct,
    trend,
    total_orders
FROM mom_comparison
WHERE prev_month_sales IS NOT NULL
ORDER BY category, `year_month`;
